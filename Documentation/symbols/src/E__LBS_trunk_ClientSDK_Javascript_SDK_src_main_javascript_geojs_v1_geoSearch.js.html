1 /**

  8 /**

 10  * @namespace
 11  */
 12 GEOAPIS_V1_INHERIT(GEOAPIS_V1.baseService, GEOAPIS_V1.geoSearch);
 13 /**
 14  * Set API URL to get location
 15  * @param Object {searchText: string(required), latitude: float(required), longitude: float(required), maxCandidates: number(optional), searchRadius: number(optional), searchRadiusUnit: string(optional)}
 16  * @param String callback method name (optional)
 17  * @return response object or calls callback
 18  */
 19 GEOAPIS_V1.geoSearch.prototype.getLocation = function(params, callback){
 20 	var apiUrl = '/geosearch/v1/locations?searchText='+encodeURIComponent(params.searchText)+'&latitude='+encodeURIComponent(params.latitude)+'&longitude='+encodeURIComponent(params.longitude);
 21 	if(params.maxCandidates !== null && params.maxCandidates !== undefined){
 22 		apiUrl += '&maxCandidates='+encodeURIComponent(params.maxCandidates);
 23 	}
 24 	if(params.searchRadius !== null && params.searchRadius !== undefined){
 25 		apiUrl += '&searchRadius='+encodeURIComponent(params.searchRadius);
 26 	}
 27 	if(params.searchRadiusUnit !== null && params.searchRadiusUnit !== undefined){
 28 		apiUrl += '&searchRadiusUnit='+encodeURIComponent(params.searchRadiusUnit);
 29 	}
 30 	if(callback !== undefined){
 31 		this.callApiAsync(apiUrl, callback);
 32 	}
 33 	else{
 34 		this.callApi(apiUrl);
 35 		return this.response;
 36 	}
 37 };
