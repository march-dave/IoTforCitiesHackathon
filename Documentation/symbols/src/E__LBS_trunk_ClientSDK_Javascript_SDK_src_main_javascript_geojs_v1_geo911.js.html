 13 /**
 14  * Set API URL to get PSAP by address
 15  * @param Object {address: free-form text} (required)
 16  * @param String callback method name (optional)
 17  * @return response object or calls callback
 18  */
 19 GEOAPIS_V1.geo911.prototype.getPSAPByAddress = function(params, callback){
 20 	var apiUrl = '/geo911/v1/psap/byaddress?address='+encodeURIComponent(params.address);
 21 	if(callback !== undefined){
 22 		this.callApiAsync(apiUrl, callback);
 23 	}
 24 	else{
 25 		this.callApi(apiUrl);
 26 		return this.response;
 27 	}
 28 };
